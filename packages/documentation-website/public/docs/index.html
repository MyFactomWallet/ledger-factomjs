signPersonalMessage(path: string, messageHex: string): Promise<{v: number, s: string, r: string}>
Parameters
path (string)
messageHex (string)
Returns
Promise<{v: number, s: string, r: string}>
Example

var v = result['v'] - 27;
v = v.toString(16);
if (v.length < 2) {
v = "0" + v;
}
console.log("Signature 0x" + result['r'] + result['s'] + v);
